What is Bootstrap-Swapbox?

A bootstrap-extension that converts a standard select box into a select box with a twist.

The twist is that while other select boxes leave the dropdown options the same after you choose one, this select box will hide the selected option from the list.

I know, not that exciting... but to see similar functionality in the wild, checkout the News Feed filter button on the Facebook Mobile App on iOS. This is what I was going for.

Usage / Demos

Basic Example


<select id="swapbox-demo-1" name="choice-toggle">
    <option value="all">All Upcoming Events</option>
    <option value="live">Upcoming Live Events</option>
    <option value="online">Online Courses</option>
</select>

<script>
    $(function() {
        $("#swapbox-demo-1").swapbox();
    });
</script>

Dynamic Append / Prepend Overrides


<select id="swapbox-demo-2" name="choice-toggle">
    <option data-icon-class="fa fa-book" value="all">All Upcoming Events</option>
    <option data-icon-class="fa fa-calendar" value="live">Upcoming Live Events</option>
    <option data-icon-class="fa fa-cloud" value="online">Online Courses</option>
</select>

<script>
    $(function() {
        $("#swapbox-demo-2").swapbox({
            itemPrepend: function(item) {
                return "<span class='" + item.$option.attr('data-icon-class') + "'></span> ";
            },
            itemAppend: function() {
                return "";
            }
        });
    });
</script>

Binding to Events

Event Console

Event messages created by the swapbox below will be presented here. 

<select id="swapbox-demo-3" name="swapbox-3">
    <option value="all">All Upcoming Events</option>
    <option value="live">Upcoming Live Events</option>
    <option value="online">Online Courses</option>
</select>

<script>
    $(function() {
        var $console = $(".demo3-console > .panel-body > .well");
        var $eventCount = 0;
        var addMessage = function(msg) {
            $eventCount++
            $console.prepend("<div>[" + $eventCount + "] " + msg + "</div>");
        }
        $("#swapbox-demo-3").swapbox();
        
        //NEW EVENTS IN BOOTSTRAP-SWAPBOX
        $("#swapbox-demo-3").on("change.bs.swapbox", function(e) {
            addMessage("changed: " + $(this).val());
        });

        //PIGGY BACKED EVENTS FROM BOOTSTRAP DROPDOWN
        //show event
        $("#swapbox-demo-3").on("show.bs.swapbox", function(e) {
            addMessage("show");
        });

        //shown event
        $("#swapbox-demo-3").on("shown.bs.swapbox", function(e) {
            addMessage("shown");
        });

        //hide event
        $("#swapbox-demo-3").on("hide.bs.swapbox", function(e) {
            addMessage("hide");
        });

        //hidden event
        $("#swapbox-demo-3").on("hidden.bs.swapbox", function(e) {
            addMessage("hidden");
        });
    });
</script>

With Facebook-like Native iOS App Look / Animation


$(function() {
    var $myCustomBox = $("#swapbox-demo-4");
    
    $myCustomBox.swapbox({
        wrapCssClass: "my-custom-swapbox",
        itemPrepend: function(item) {
            return "<span class='" + item.$option.attr('data-icon-class') + "'></span> ";
        },
        itemAppend: function() {
            return "";
        }
    });

    $myCustomBox
        .on("show.bs.swapbox", function(e) {
            var $swapboxWrap = $(e.relatedTarget);
            var $list = $swapboxWrap.find('.swapbox-list').first().stop(true,true);

            $swapboxWrap.addClass('opening');
            $list.slideDown( 250 , function() {
                $swapboxWrap.removeClass('opening');
            });
            //$swapboxWrap.delay(2000).removeClass('opening');
            //$(this).find('.dropdown')
        })
        .on("shown.bs.swapbox", function(e) {
            var $swapboxWrap = $(e.relatedTarget);
            //var $list = $swapboxWrap.find('.swapbox-list').first().stop(true,true);

            //$swapboxWrap.delay(250).removeClass('opening');
            //$list.slideDown();
            //$(this).find('.dropdown')
        })
        .on("hide.bs.swapbox", function(e) {
            var $swapboxWrap = $(e.relatedTarget);
            $swapboxWrap.addClass('closing')
            var $list = $swapboxWrap.find('.swapbox-list').first().stop(true,true);
            
            $list.slideUp(250, function() {
                $swapboxWrap.removeClass('closing')
            });
            //$(this).find('.dropdown')
        });
});
